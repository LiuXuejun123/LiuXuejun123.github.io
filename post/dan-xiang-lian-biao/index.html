2020-11-10 / 6 min read

节点类：

/*节点类*/
class Node{
public:
    int data;//数据
    Node* next;//指针
    /*构造*/
    Node() :data(0),next(nullptr){};
    Node(int data): data(data),next(nullptr){};
    /*析构*/
    ~Node(){};
};

链表类

class Linklist{
private:
    Node* head;
};

链表中主要的就是头指针。其内部成员函数受链表结构影响 稍有不同，详见如下

1.带头节点


特点 ：头指针指向一个节点，但这个节点不存储数据，真正的链表节点从1号节点开始。头指针中存储的是头节点的地址。

class Linklist{
public:
    Linklist();//构造

    void DeleteElemAtEnd();       //在尾部删除元素

    void DeleteElemAtHead();      //在头部删除节点
    Node* reserverlinklist();     //反转链表
private:
    Node* head;
};

获取链表长度int GetLength();
int Linklist::GetLength() {//获得链表长度
    int size=0;//计数
    Node* p=head->next;//
    while(p!= nullptr)
    {
        size++;
        p=p->next;
    }
    return size;
}

Node* Linklist::FindInList(int data) {
    Node * p = head;
    if (p==nullptr)
    {
        cout<<"The Linklist is empty"<<endl;
        return nullptr;
    }
    else
    {
        while(p->next!= nullptr)
        {
            if(p->data == data)
            {
                return p;
            }
            p=p->next;
        }
    }
    if(p->data == data)//头节点也有数据
    {
        return p;
    }
    else
    {
        return nullptr;
    }
}

void Linklist::InsertElemAtEnd(int data)          //在尾部插入指定的元素
{
    Node* newnode=new Node;
    newnode->next= nullptr;
    newnode->data=data;
    Node* p=head;
    if(head->next== nullptr)
    {
        head->next=newnode;
    }
    while(p->next!= nullptr)
    {
        p=p->next;
    }
    p->next=newnode;
}

头部添加 InsertElemAtHead（int data）
void Linklist::InsertElemAtHead(int data)    //在头部插入指定元素
{
    Node* newnode=new Node;
    newnode->data=data;
    Node* p=head->next;
    if(head== nullptr)
    {
        head=newnode;
    }
    newnode->next=p->next;
    p->next=newnode;
}

p指向头节点的地址，p->next 是第一个节点的地址
